import { render } from 'solid-js/web'
import { createSignal } from 'solid-js'

function App() {
  const [count, setCount] = createSignal(0)

  return (
    <div class="container">
      <h1>Solid.js + Pen</h1>
      <p>Count: {count()}</p>
      <button onClick={() => setCount(c => c + 1)}>Increment</button>
    </div>
  )
}

render(() => <App />, document.getElementById('app'))
